--- src/store/content.tsx ---
import React, { createContext, useContext, useState, useEffect } from 'react';
import { SiteContent } from '../domain/types';
import { DEFAULT_CONTENT } from '../data/default-content';
import { saveToStorage, getFromStorage } from '../services/storage';

const ContentContext = createContext<{ content: SiteContent; updateContent: (c: SiteContent) => void }>(null!);

export const ContentProvider: React.FC<{children: React.ReactNode}> = ({ children }) => {
  const [content, setContent] = useState<SiteContent>(() => getFromStorage('content') || DEFAULT_CONTENT);
  const updateContent = (newContent: SiteContent) => {
    setContent(newContent);
    saveToStorage('content', newContent);
  };
  return <ContentContext.Provider value={{ content, updateContent }}>{children}</ContentContext.Provider>;
};
export const useContent = () => useContext(ContentContext);

--- src/App.tsx ---
import { BrowserRouter, Routes, Route } from 'react-router-dom';
import { AuthProvider } from './store/auth';
import { ContentProvider } from './store/content';
import HomePage from './pages/public/HomePage';
import DashboardPage from './pages/admin/DashboardPage';
import ProtectedRoute from './router/ProtectedRoute';

export default function App() {
  return (
    <AuthProvider>
      <ContentProvider>
        <BrowserRouter>
          <Routes>
            <Route path="/" element={<HomePage />} />
            <Route path="/admin" element={<ProtectedRoute><DashboardPage /></ProtectedRoute>} />
          </Routes>
        </BrowserRouter>
      </ContentProvider>
    </AuthProvider>
  );
}

--- README.md ---
# Posto Via 14
Projeto React + Vite + TS para gestão de conteúdo de posto de combustíveis.

## Arquitetura
- Context API para estado global (Auth e Conteúdo).
- localStorage para persistência.
- Roteamento com React Router v6.
- CSS puro para estilização.

## Como rodar
1. npm install
2. npm run dev
